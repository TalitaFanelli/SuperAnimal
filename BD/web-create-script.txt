CREATE TABLE CLIENTE (
  CPF VARCHAR(14) NOT NULL,
  SENHA VARCHAR(8) NOT NULL,
  PRIMARY KEY(CPF)
);

CREATE TABLE ANIMAL (
  ID_ANIMAL INTEGER UNSIGNED NOT NULL AUTO_INCREMENT,
  CPF_CLI_MED VARCHAR(14) NOT NULL,
  NOME VARCHAR(25) NOT NULL,
  SOBRENOME VARCHAR(45) NOT NULL,
  NASCIMENTO VARCHAR(10) NOT NULL,
  SEXO CHAR(1) NOT NULL,
  ESPECIE VARCHAR(11) NOT NULL,
  RACA VARCHAR(25) NOT NULL,
  PELAGEM VARCHAR(20) NOT NULL,
  PRIMARY KEY(ID_ANIMAL)
);

CREATE TABLE MEDICO (
  CPF VARCHAR(14) NOT NULL,
  CRMV INTEGER(6) UNSIGNED NOT NULL,
  SENHA VARCHAR(8) NOT NULL,
  PRIMARY KEY(CPF)
);

CREATE TABLE DADOS_PESSOAIS_MEDICO (
  ID_DADOS_PESSOAIS INTEGER UNSIGNED NOT NULL AUTO_INCREMENT,
  MEDICO_CPF VARCHAR(14) NOT NULL,
  NOME VARCHAR(40) NOT NULL,
  SOBRENOME VARCHAR(45) NOT NULL,
  NASCIMENTO VARCHAR(10) NOT NULL,
  GENERO CHAR(1) NOT NULL,
  PRIMARY KEY(ID_DADOS_PESSOAIS),
  FOREIGN KEY(MEDICO_CPF)
    REFERENCES MEDICO(CPF)
      ON DELETE CASCADE
      ON UPDATE NO ACTION
);

CREATE TABLE DADOS_RESIDENCIAIS_CLIENTE (
  ID_DADOS_RESIDENCIAIS INTEGER UNSIGNED NOT NULL AUTO_INCREMENT,
